# React's new syntax sugar

Let's change `.jsx` to `.js` right now.

### @legacy
```jsx
<>
  <p className="text"><span>bbb</span>aaa</p>
  <butoon onClick={alert}>button</button>
  jsx
</>
```

### @new
```js
$.Fragment({}, $.p({
	className: "text"
}, $.span({}, "bbb"), "aaa"), $.button({
	onClick: alert
}, "button"), "jsx")
```

## # How to syntax?

### @Tag
`$.{tagName}({attributes}, children...)`

**example**

*FReact*
```js
$.p({className: "text"}, "Hello world!")
```

*JSX*
```html
<p className="text">
  Hello world!
</p>
```

### @Component
`{ComponentName}({...props})`

**example**

*FReact*
```js
Box({color: "red"})
```

*JSX*
```html
<Box color="red" />
```


**if component need children node in component.**

`{ComponentName}({...props, children: {children}})`

**example**

*FReact*
```js
Box({color: "red", children: $.p({}, "Hello React!")})
```

*JSX*
```html
<Box color="red">
  Hello React!
</Box>
```

import { useState } from "preact/hooks";
import { $ } from "https://deno.land/x/freact/mod.ts";

export default function Counter() {
  const [ count, setCount ] = useState(0);
  
  $.create("countNumber");

  return $.Fragment({}, 
    $.countNumber({}, count),
    $.button({ onClick: setCount(count++) }, "+")
  )
}
